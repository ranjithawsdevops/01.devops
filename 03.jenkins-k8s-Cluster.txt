pipeline {
    agent none
    options {
        timeout(time: 1,unit: 'HOURS')
    }
     stages {
        stage('Checkout')
		{
            agent { label 'slave' }
            steps {
				    git branch: 'main', 
				    credentialsId: 'gitlab_user',
				    url: 'https://github.com/ranjithawsdevops/jenkins-k8s-cluster.git'

            }
        }
    
		stage('PreCheck')
		{
		    agent { label 'slave' }
			   
	    	steps {
		      script {
		       env.BUILDME = "yes" //if yes then run below
		  
		    }
		}
		}
     	
        stage('Deploy Terraform') 
          {
          agent { label 'slave' }
			when {environment name: 'BUILDME', value: 'yes'}
			steps {

			  script {
			    dir('dev') {
	      
			        sh 'ls'
			        sh 'terraform init'
                                       sh 'terraform plan'
                                       sh 'terraform apply --auto-approve'
			    }
			  }
      
    }
  }
		 
    }
}
